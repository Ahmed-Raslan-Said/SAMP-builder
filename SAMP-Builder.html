}

if(data.conditionNotes) document.getElementById('condition-notes').value = data.conditionNotes;
if(data.cannotDo) document.getElementById('cannot-do').value = data.cannotDo;

} catch(e) { return false; }
}

function validateBeforeGenerate() {
const errors = [];
if(!document.getElementById('org-name')?.value) errors.push('Organization name is required (Module 1)');
if(!document.getElementById('org-vision')?.value) errors.push('Vision statement is required (Module 1)');
if(!document.getElementById('org-period')?.value) errors.push('Plan period is required (Module 1)');

const hasAssets = state.assetClasses.some(ac => {
const row = document.getElementById('arow-'+state.assetClasses.indexOf(ac));
if(!row) return ac.grc > 0;
const inputs = row.querySelectorAll('input[type="number"]');
return parseFloat(inputs[0]?.value) > 0;
});
if(!hasAssets) errors.push('At least one asset class with GRC value is required (Module 2)');

if(errors.length > 0) {
showToast('⚠ ' + errors[0] + (errors.length > 1 ? ' (+' + (errors.length-1) + ' more)' : ''), 'error', 4000);
return false;
}
return true;
}

function injectPopovers() {
const targets = [

const label = el.previousElementSibling;
if(!label || !label.classList.contains('field-label')) return;
if(label.querySelector('.pop-btn')) return; // already added

const tip = HELP_TIPS[t.key];
if(!tip) return;

const btn = document.createElement('button');
btn.className = 'pop-btn';
btn.textContent = '?';
btn.setAttribute('type','button');
btn.addEventListener('mouseenter', () => showPop(t.key));
btn.addEventListener('mouseleave', () => hidePop(t.key));

const pop = document.createElement('div');
pop.className = 'popover';
pop.id = 'pop-'+t.key;
pop.innerHTML = `<div class="pop-title">${tip.title}</div>

label.appendChild(btn);
const wrap = el.parentElement;
if(wrap) { wrap.style.position='relative'; wrap.appendChild(pop); }
});
}
